AP_Annees <- WDPA_Senegal %>%
  filter(st_geometry_type(.) == "MULTIPOLYGON") %>% 
  mutate(GIS_AREA= st_area(.), 
         GIS_AREA= units::set_units(GIS_AREA, "km2"))%>% 
  group_by(STATUS_YR) %>%
  summarise(n = n(),
            area = sum(GIS_AREA)) %>%
  arrange(STATUS_YR)
